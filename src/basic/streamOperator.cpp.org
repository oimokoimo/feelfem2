#include <iostream>
#include "string.hpp"

std::ostream& operator<<(std::ostream& os, const string& s)
{
    for (std::size_t i = 0; i < s.length(); ++i)
    {
        os.put(s[i]);
    }
    return os;
}

std::istream& operator>>(std::istream& is, string& s)
{
    std::string buffer;
    is >> buffer;

    if (is)
    {
        s = buffer.c_str();
    }

    return is;
}
